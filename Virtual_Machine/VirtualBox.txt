====== VirtualBox ======

VBoxManage list vms
VBoxManage list ostypes
VBoxManage guestproperty enumerate winxp01
VBoxManage startvm winxp01 --type gui
VBoxManage startvm winxp01 --type headless


VBoxManage createvm --name "winxp01" --ostype "WindowsXP" --register
VBoxManage createhd --filename "/home/houbinyi/winxp.vdi" --size 10 --format vdi --variant standard
VBoxManage storagectl winxp01 --name "IDE Controller" --add ide --controller PIIX3
VBoxManage stroagetattach winxp01 --storagectl "IDE COntroller" --port 0 --device 0 --type hdd --medium "/home/houbinyi/winxp.vdi"
VBoxManage storagectl winxp01 --name "SATA Controller" --add sata --controller IntelAHCI
VBoxManage stroagetattach winxp01 --storagectl "SATA Controller" --port 1 --device 0 --type dvddrive --medium "/home/houbinyi/winxp.install.iso"
VBoxManage showvminfo winxp01
